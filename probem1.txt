#include <iostream> 
using namespace std; 
  

string maximizeNumber(string N, int M) 
{ 
   
    sort(N.begin(), N.end()); 
  
    for (int i = 0; i < N.size(); i++) 
        for (int j = i + 1; j < N.size(); j++) { 
  
            string t = N; 
 
            swap(t[j], t[i]); 
  
          
            sort(t.begin() + i + 1, t.end()); 

            if (stoll(t) > stoll(N) && stoll(t) <= M) 
              

                swap(N[i], N[j]); 
        } 
  
  
    return N; 
}
